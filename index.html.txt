    ctx.moveTo(10, -30); ctx.lineTo(20, -40); ctx.moveTo(-10, -30); ctx.lineTo(-20, -40); ctx.stroke();
  } else if(cls.name === 'Pyromancien'){
    for(let i=0;i<3;i++){const fy=-45-i*5+Math.sin(Date.now()*0.01+i)*2;const fr=4-i*0.8;ctx.fillStyle=i===0?'#ffaa00':'#ff4400';ctx.shadowColor='#ffaa00';ctx.beginPath();ctx.arc(15,fy,fr,0,Math.PI*2);ctx.fill();}
  } else if(cls.name === 'Maël'){
    // Katana Maël
    const hue = (Date.now()/5)%360;
    ctx.shadowBlur = 25;
    ctx.shadowColor = `hsl(${hue}, 100%, 50%)`;
    ctx.strokeStyle = `hsl(${hue}, 100%, 50%)`;
    ctx.lineWidth = 4;
    ctx.beginPath();
    if(attacking>0){
      const ang = -Math.PI/2 + (1-attacking)*1.5;
      ctx.moveTo(10, -25);
      ctx.lineTo(10 + Math.cos(ang)*50, -25 + Math.sin(ang)*50);
    } else {
      ctx.moveTo(12, -25); ctx.lineTo(12, -60); // katana pointing up
    }
    ctx.stroke();
    // Garde du katana
    ctx.lineWidth = 6;
    ctx.beginPath();
    ctx.moveTo(8, -25); ctx.lineTo(16, -25);
    ctx.stroke();
  }
  
  if(shielding){
    ctx.shadowBlur = 20; ctx.shadowColor = '#00f0ff';
    ctx.strokeStyle = 'rgba(0,240,255,0.6)'; ctx.lineWidth = 2;
    ctx.beginPath(); ctx.arc(0, -20, 25, -Math.PI/2, Math.PI/2); ctx.stroke();
  }
  ctx.restore();
}

function handleNetData(data){
  if(data.type === 'select'){
    if(netRole === 'host'){ p2Choice = data.cls; updatePreviews(); updateCharGrid(); checkFightBtn(); }
    else { p1Choice = data.cls; updatePreviews(); updateCharGrid(); checkFightBtn(); }
  } else if(data.type === 'start'){
    startGame();
  } else if(data.type === 'inputs'){
    // Sync inputs from client
    if(netRole === 'host'){
      KEYS['arrowleft'] = data.keys.left;
      KEYS['arrowright'] = data.keys.right;
      KEYS['arrowup'] = data.keys.up;
      KEYS['arrowdown'] = data.keys.down;
      KEYS['k'] = data.keys.k;
      KEYS['l'] = data.keys.l;
      KEYS['m'] = data.keys.m;
    }
  } else if(data.type === 'state'){
    // Sync state from host
    if(netRole === 'client'){
      // Mapping pour s'assurer que la classe est bien liée aux joueurs côté client
      players = data.state.players.map(p => ({...p, cls: CLASSES[p._clsKey]}));
      projectiles = data.state.projectiles;
      cameraShake = data.state.cameraShake;
      frameCount = data.state.frameCount;
      updateHUD();
    }
  }
}

function gameLoop(){
  if(!gameRunning) return;
  
  if(netRole === 'host' || netRole === 'local'){
    update();
    if(netRole === 'host' && conn && conn.open){
      sendNet({type:'state', state:{
        players: players.map(p=>({
          x:p.x, y:p.y, vx:p.vx, vy:p.vy, facing:p.facing, percent:p.percent, lives:p.lives, 
          attackAnim:p.attackAnim, shielding:p.shielding, hurtTimer:p.hurtTimer, walkPhase:p.walkPhase, 
          active:p.active, invuln:p.invuln, _clsKey:p._clsKey, charging:p.charging, dashing:p.dashing, flaming:p.flaming,
          specialCooldown: p.specialCooldown, special2Cooldown: p.special2Cooldown // FIX: Envoi des cooldowns
        })),
        projectiles: projectiles,
        cameraShake: cameraShake,
        frameCount: frameCount
      }});
    }
  } else if(netRole === 'client'){
    sendNet({type:'inputs', keys:{
      left: KEYS['arrowleft'], right: KEYS['arrowright'], up: KEYS['arrowup'], down: KEYS['arrowdown'],
      k: KEYS['k'], l: KEYS['l'], m: KEYS['m']
    }});
  }
  
  draw();
  requestAnimationFrame(gameLoop);
}

function update(){
  frameCount++;
  players.forEach(p=>{
    if(!p.active){ p.respawnTimer--; if(p.respawnTimer<=0) respawnPlayer(p); return; }
    if(p.invuln>0) p.invuln--; if(p.hurtTimer>0) p.hurtTimer--;
    if(p.attackCooldown>0) p.attackCooldown--; if(p.specialCooldown>0) p.specialCooldown--;
    if(p.special2Cooldown>0) p.special2Cooldown--;
    if(p.attackAnim>0) p.attackAnim -= 0.08; if(p.attackAnim<0) p.attackAnim=0;
    
    if(p.charging>0){
      p.charging--; p.vx = p.chargingDir * 12;
      const other = players.find(o=>o.id!==p.id && o.active);
      if(other && Math.abs(other.x-p.x)<50 && Math.abs(other.y-p.y)<60){ if(other.invuln<=0 && !other.shielding) damagePlayer(other, p.cls.specialDmg, p.chargingDir*p.cls.specialKB*2, -3); }
    }
    if(p.dashing>0){
      p.dashing--; p.vx = p.facing * 16; p.vy = 0;
      if(frameCount%2===0) particles.push({x:p.x,y:p.y-30,vx:0,vy:0,life:20,maxLife:20,color:p.cls.color,size:6,type:'trail'});
      const other = players.find(o=>o.id!==p.id && o.active);
      if(other && Math.abs(other.x-p.x)<50 && Math.abs(other.y-p.y)<60 && p.dashing>15){ if(other.invuln<=0 && !other.shielding){ damagePlayer(other, p.cls.specialDmg, p.facing*p.cls.specialKB, -6); p.dashing = 0; } }
    }
    if(p.flaming>0){
      p.flaming--;
      for(let i=0;i<3;i++) particles.push({x:p.x+p.facing*20+(Math.random()-0.5)*10,y:p.y-30+(Math.random()-0.5)*20,vx:p.facing*4+(Math.random()-0.5)*2,vy:-1+Math.random()*2,life:30,maxLife:30,color:Math.random()>0.5?'#ffaa00':'#ff4400',size:5+Math.random()*3,type:'fire'});
      const other = players.find(o=>o.id!==p.id && o.active);
      if(other && other.invuln<=0){ const dx=other.x-p.x; if((p.facing>0&&dx>0&&dx<120)||(p.facing<0&&dx<0&&dx>-120)){ if(Math.abs(other.y-p.y)<60){ if(frameCount%10===0 && !other.shielding) damagePlayer(other, 3, p.facing*2, -2); } } }
    }
    
    const c = p.controls;
    const leftPressed = KEYS[c.left]; const rightPressed = KEYS[c.right];
    
    if(p.charging<=0 && p.dashing<=0){
      p.shielding = KEYS[c.down] && p.onGround;
      if(!p.shielding){
        if(leftPressed){ p.vx -= 0.8; p.facing = -1; p.walkPhase += 0.25; }
        if(rightPressed){ p.vx += 0.8; p.facing = 1; p.walkPhase += 0.25; }
        if(!leftPressed && !rightPressed){ p.vx *= FRICTION; if(Math.abs(p.vx)<0.1) p.vx=0; }
        
        // Saut (Maël a des sauts infinis)
        const maxJumps = p.cls.name === 'Maël' ? 999 : p.maxJumps;
        if(KEYS[c.up] && !p._jumpHeld){
          if(p.jumps < maxJumps){
            p.vy = -p.cls.jumpPower; p.jumps++;
            for(let i=0;i<6;i++) particles.push({x:p.x,y:p.y,vx:(Math.random()-0.5)*3,vy:Math.random()*2,life:15,maxLife:15,color:p.cls.color,size:3,type:'dust'});
          }
          p._jumpHeld = true;
        }
        if(!KEYS[c.up]) p._jumpHeld = false;
        
        if(KEYS[c.attack] && p.attackCooldown<=0 && !p._atkHeld){ performAttack(p); p._atkHeld = true; }
        if(!KEYS[c.attack]) p._atkHeld = false;
        
        if(KEYS[c.special] && p.specialCooldown<=0 && !p._spHeld){ performSpecial(p); p._spHeld = true; }
        if(!KEYS[c.special]) p._spHeld = false;
        
        // Spécial 2 pour Maël
        if(p.cls.name === 'Maël' && KEYS[c.special2] && p.special2Cooldown<=0 && !p._sp2Held){ performSpecial2(p); p._sp2Held = true; }
        if(!KEYS[c.special2]) p._sp2Held = false;
      } else { p.vx *= 0.5; }
    }
    
    const maxSpeed = p.cls.speed;
    if(p.vx > maxSpeed) p.vx = maxSpeed; if(p.vx < -maxSpeed) p.vx = -maxSpeed;
    if(p.dashing<=0){ p.vy += GRAVITY; if(p.vy > 18) p.vy = 18; }
    
    p.x += p.vx; p.y += p.vy;
    
    p.onGround = false;
    platforms.forEach(plat=>{
      if(p.x + p.width/2 > plat.x && p.x - p.width/2 < plat.x + plat.w){
        if(p.y > plat.y && p.y - p.vy <= plat.y + 5 && p.vy >= 0){ p.y = plat.y; p.vy = 0; p.onGround = true; p.jumps = 0; }
      }
    });
    
    if(p.x < -100 || p.x > ARENA_W + 100 || p.y > ARENA_H + 100) killPlayer(p);
    if(p.y < -200) p.vy = 2;
  });
  
  projectiles = projectiles.filter(pr=>{
    pr.x += pr.vx; pr.y += pr.vy; pr.life--;
    if(pr.gravity) pr.vy += 0.2;
    if(pr.type === 'fireball') particles.push({x:pr.x,y:pr.y,vx:0,vy:0,life:15,maxLife:15,color:Math.random()>0.5?'#ffaa00':'#ff4400',size:4,type:'trail'});
    
    let hit = false;
    players.forEach(p=>{
      if(p.id === pr.owner || !p.active || p.invuln>0) return;
      if(Math.abs(p.x - pr.x) < 25 && Math.abs(p.y - 30 - pr.y) < 35){
        if(!p.shielding) damagePlayer(p, pr.dmg, pr.kb * Math.sign(pr.vx || 1), pr.kb * -0.3);
        hit = true;
        for(let i=0;i<12;i++) particles.push({x:pr.x,y:pr.y,vx:(Math.random()-0.5)*6,vy:(Math.random()-0.5)*6,life:25,maxLife:25,color:pr.color||'#fff',size:3+Math.random()*2,type:'spark'});
      }
    });
    if(hit) return false;
    if(pr.type !== 'fireball'){ platforms.forEach(plat=>{ if(pr.x>plat.x&&pr.x<plat.x+plat.w&&pr.y>plat.y&&pr.y<plat.y+plat.h) hit=true; }); }
    return !hit && pr.life > 0 && pr.x > -50 && pr.x < ARENA_W+50 && pr.y < ARENA_H+50;
  });
  
  particles = particles.filter(pt=>{ pt.x+=pt.vx; pt.y+=pt.vy; if(pt.type==='fire'||pt.type==='spark') pt.vy+=0.1; if(pt.type==='dust') pt.vy+=0.2; pt.life--; return pt.life>0; });
  if(cameraShake > 0) cameraShake *= 0.9;
  
  updateHUD();
  
  const alive = players.filter(p=>p.lives>0);
  if(alive.length <= 1 && frameCount > 60) endGame(alive[0]);
}

function performAttack(p){
  p.attackCooldown = p.cls.name === 'Maël' ? 10 : 25;
  p.attackAnim = 1;
  const other = players.find(o=>o.id!==p.id && o.active);
  if(other && other.invuln<=0){
    const dx = other.x - p.x; const dy = other.y - p.y;
    if(Math.abs(dx) < p.cls.attackRange && Math.abs(dy) < 60 && Math.sign(dx) === p.facing){
      if(other.shielding) damagePlayer(other, p.cls.attackDmg*0.3, p.facing*p.cls.attackKB*0.3, -1);
      else damagePlayer(other, p.cls.attackDmg, p.facing*p.cls.attackKB, -2);
    }
  }
  for(let i=0;i<6;i++) particles.push({x:p.x+p.facing*30,y:p.y-30,vx:p.facing*3+(Math.random()-0.5)*2,vy:(Math.random()-0.5)*3,life:12,maxLife:12,color:p.cls.color,size:3,type:'spark'});
}

function performSpecial(p){
  p.specialCooldown = p.cls.specialCD;
  const other = players.find(o=>o.id!==p.id && o.active);
  
  switch(p.cls.name){
    case 'Épéiste':
      p.attackAnim = 1.5;
      for(let i=0;i<20;i++){const ang=(i/20)*Math.PI*2;particles.push({x:p.x+Math.cos(ang)*40,y:p.y-30+Math.sin(ang)*40,vx:Math.cos(ang)*4,vy:Math.sin(ang)*4,life:25,maxLife:25,color:p.cls.color,size:4,type:'spark'});}
      if(other&&Math.abs(other.x-p.x)<80&&Math.abs(other.y-p.y)<80){if(other.invuln<=0&&!other.shielding){const kbDir=Math.sign(other.x-p.x)||p.facing;damagePlayer(other,p.cls.specialDmg,kbDir*p.cls.specialKB,-5);}}
      cameraShake=8; break;
    case 'Archer':
      for(let i=0;i<3;i++)setTimeout(()=>{if(p.active)projectiles.push({x:p.x,y:p.y-35,vx:p.facing*12,vy:-2+i*1.5,life:80,type:'arrow',dmg:p.cls.specialDmg,kb:p.cls.specialKB,owner:p.id,color:'#88ff88',gravity:true});},i*120); break;
    case 'Mage':
      projectiles.push({x:p.x,y:p.y-35,vx:p.facing*9,vy:0,life:120,type:'fireball',dmg:p.cls.specialDmg,kb:p.cls.specialKB,owner:p.id,color:'#ff6600',gravity:false});
      for(let i=0;i<15;i++)particles.push({x:p.x,y:p.y-35,vx:(Math.random()-0.5)*4,vy:(Math.random()-0.5)*4,life:20,maxLife:20,color:'#ffaa00',size:4,type:'spark'}); break;
    case 'Tank':
      p.charging=30; p.chargingDir=p.facing; cameraShake=5; break;
    case 'Assassin':
      if(other){p.x=other.x-p.facing*40; p.dashing=25; for(let i=0;i<20;i++)particles.push({x:p.x,y:p.y-30,vx:(Math.random()-0.5)*8,vy:(Math.random()-0.5)*8,life:20,maxLife:20,color:p.cls.color,size:3,type:'spark'});} else p.dashing=25; break;
    case 'Pyromancien':
      p.flaming=60; break;
    case 'Maël':
      // Iaido : Téléportation derrière l'adversaire et slash
      p.attackAnim = 2;
      if(other){
        p.x = other.x - p.facing * 50;
        for(let i=0;i<30;i++)particles.push({x:p.x,y:p.y-30,vx:(Math.random()-0.5)*10,vy:(Math.random()-0.5)*10,life:25,maxLife:25,color:p.cls.color,size:4,type:'spark'});
        if(other.invuln<=0 && !other.shielding) damagePlayer(other, p.cls.specialDmg, p.facing*p.cls.specialKB, -10);
      }
      cameraShake = 15;
      break;
  }
}

function performSpecial2(p){
  p.special2Cooldown = p.cls.special2CD;
  const other = players.find(o=>o.id!==p.id && o.active);
  
  if(p.cls.name === 'Maël'){
    // Exécution Divine : One-shot
    p.attackAnim = 3;
    cameraShake = 40;
    
    // Effet visuel gigantesque
    for(let i=0;i<80;i++){
      const ang = Math.random()*Math.PI*2;
      const speed = Math.random()*15+5;
      particles.push({x:p.x, y:p.y-30, vx:Math.cos(ang)*speed, vy:Math.sin(ang)*speed, life:40, maxLife:40, color:Math.random()>0.5?'#ffd700':'#ff0000', size:5+Math.random()*4, type:'spark'});
    }
    
    if(other){
      // Touche toujours, peu importe la distance
      other.percent = 999;
      // Knockback monumental vers le haut et le côté
      const kbDir = Math.sign(other.x - p.x) || p.facing;
      damagePlayer(other, 0, kbDir * 50, -60);
      // Force le kill direct
      setTimeout(()=>{
        if(other.active) killPlayer(other);
      }, 100);
    }
  }
}

function damagePlayer(p, dmg, kbX, kbY){
  p.percent += dmg;
  const kbMult = 1 + p.percent/50;
  p.vx += kbX * kbMult / p.cls.weight;
  p.vy += kbY * kbMult / p.cls.weight;
  p.hurtTimer = 12; p.invuln = 8;
  cameraShake = Math.min(20, cameraShake + dmg*0.3 + (p.cls.name==='Maël'?5:0));
  for(let i=0;i<8;i++)particles.push({x:p.x,y:p.y-30,vx:(Math.random()-0.5)*5,vy:-Math.random()*4,life:20,maxLife:20,color:'#ff4466',size:3,type:'spark'});
}

function killPlayer(p){
  if(!p.active) return;
  p.lives--; p.active=false; p.respawnTimer=90;
  for(let i=0;i<40;i++){const ang=(i/40)*Math.PI*2;particles.push({x:p.x,y:p.y,vx:Math.cos(ang)*8,vy:Math.sin(ang)*8,life:40,maxLife:40,color:p.cls.color,size:5,type:'spark'});}
  cameraShake = 25;
  showToast(`Joueur ${p.id} éliminé ! (${p.lives} vie${p.lives>1?'s':''} restante${p.lives>1?'s':''})`);
}

function respawnPlayer(p){
  if(p.lives <= 0) return;
  p.active=true; p.x=p.id===1?300:800; p.y=200; p.vx=0; p.vy=0; p.percent=0; p.invuln=60; p.hurtTimer=0; p.shielding=false; p.charging=0; p.dashing=0; p.flaming=0;
}

function updateHUD(){
  if(!players[0] || !players[1]) return;
  players.forEach((p, i)=>{
    const idx=i+1;
    const pctEl = document.getElementById('hudPercent'+idx);
    pctEl.textContent = Math.floor(p.percent) + '%';
    document.getElementById('hudLives'+idx).textContent = '♥'.repeat(Math.max(0,p.lives)) || '—';
    if(p.percent > 100) pctEl.style.color='#ff0044'; else if(p.percent > 50) pctEl.style.color='#ffaa00'; else pctEl.style.color=p.cls.color;
    const hpRatio = Math.max(0, 1 - p.percent/200);
    document.getElementById('hudBar'+idx).style.width = (hpRatio*100)+'%';
    
    const cdEl = document.getElementById('cd'+idx);
    let txt = `${p.cls.specialName} : `;
    if(p.specialCooldown > 0) txt += `${Math.ceil(p.specialCooldown/60)}s`; else txt += `PRÊT`;
    if(p.cls.name === 'Maël'){
      txt += ` | ${p.cls.special2Name} : `;
      if(p.special2Cooldown > 0) txt += `${Math.ceil(p.special2Cooldown/60)}s`; else txt += `PRÊT`;
    }
    cdEl.textContent = txt;
    if(p.cls.name === 'Maël'){
      cdEl.classList.toggle('cd-ready', p.specialCooldown<=0 && p.special2Cooldown<=0);
    } else {
      cdEl.classList.toggle('cd-ready', p.specialCooldown<=0);
    }
  });
}

function draw(){
  gtx.save();
  if(cameraShake > 0.1) gtx.translate((Math.random()-0.5)*cameraShake, (Math.random()-0.5)*cameraShake);
  
  const grad = gtx.createLinearGradient(0,0,0,ARENA_H);
  grad.addColorStop(0,'#0a0a14'); grad.addColorStop(0.6,'#1a1a3a'); grad.addColorStop(1,'#2a1a3a');
  gtx.fillStyle = grad; gtx.fillRect(0,0,ARENA_W,ARENA_H);
  
  gtx.fillStyle = 'rgba(255,255,255,0.6)';
  for(let i=0;i<40;i++){const sx=(i*73)%ARENA_W;const sy=(i*41)%300;const tw=Math.sin(frameCount*0.05+i)*0.5+0.5;gtx.globalAlpha=tw*0.5;gtx.fillRect(sx,sy,2,2);}
  gtx.globalAlpha = 1;
  
  platforms.forEach(plat=>{
    gtx.shadowColor='#00f0ff'; gtx.shadowBlur=15;
    gtx.fillStyle = plat.type==='main'?'#1a1a4a':'#2a2a5a';
    gtx.fillRect(plat.x, plat.y, plat.w, plat.h);
    gtx.shadowBlur=0;
    gtx.strokeStyle = plat.type==='main'?'#00f0ff':'#6688ff'; gtx.lineWidth=2;
    gtx.strokeRect(plat.x, plat.y, plat.w, plat.h);
    gtx.fillStyle = plat.type==='main'?'#00f0ff':'#88aaff';
    gtx.fillRect(plat.x, plat.y, plat.w, 2);
  });
  
  particles.forEach(pt=>{
    const a = pt.life / pt.maxLife; gtx.globalAlpha = a; gtx.fillStyle = pt.color;
    gtx.shadowColor = pt.color; gtx.shadowBlur = 10;
    if(pt.type === 'fire'){ gtx.beginPath(); gtx.arc(pt.x, pt.y, pt.size*a, 0, Math.PI*2); gtx.fill(); }
    else gtx.fillRect(pt.x-pt.size/2, pt.y-pt.size/2, pt.size, pt.size);
  });
  gtx.globalAlpha = 1; gtx.shadowBlur = 0;
  
  projectiles.forEach(pr=>{
    gtx.shadowColor = pr.color; gtx.shadowBlur = 15;
    if(pr.type === 'arrow'){
      gtx.strokeStyle = pr.color; gtx.lineWidth = 3; gtx.beginPath();
      gtx.moveTo(pr.x, pr.y); gtx.lineTo(pr.x - pr.vx*1.5, pr.y - pr.vy*1.5); gtx.stroke();
      gtx.fillStyle = '#fff'; gtx.beginPath(); gtx.arc(pr.x, pr.y, 3, 0, Math.PI*2); gtx.fill();
    } else if(pr.type === 'fireball'){
      gtx.fillStyle = '#ff4400'; gtx.beginPath(); gtx.arc(pr.x, pr.y, 12, 0, Math.PI*2); gtx.fill();
      gtx.fillStyle = '#ffaa00'; gtx.beginPath(); gtx.arc(pr.x, pr.y, 7, 0, Math.PI*2); gtx.fill();
      gtx.fillStyle = '#ffff00'; gtx.beginPath(); gtx.arc(pr.x, pr.y, 3, 0, Math.PI*2); gtx.fill();
    }
    gtx.shadowBlur = 0;
  });
  
  players.forEach(p=>{
    if(!p.active) return;
    if(p.invuln > 0 && frameCount%6 < 3) gtx.globalAlpha = 0.5;
    
    gtx.fillStyle = 'rgba(0,0,0,0.4)';
    gtx.beginPath(); gtx.ellipse(p.x, p.y+2, 18, 5, 0, 0, Math.PI*2); gtx.fill();
    
    // Make sure cls is resolved for drawing
    const drawCls = CLASSES[p._clsKey] || p.cls;
    drawStickman(gtx, p.x, p.y, drawCls, {facing:p.facing,walkPhase:p.walkPhase,attacking:p.attackAnim,shielding:p.shielding,hurt:p.hurtTimer});
    
    gtx.fillStyle = drawCls.color; gtx.font = 'bold 14px Orbitron'; gtx.textAlign = 'center';
    gtx.shadowColor = drawCls.color; gtx.shadowBlur = 10;
    gtx.fillText('P'+p.id, p.x, p.y-55);
    gtx.shadowBlur = 0; gtx.globalAlpha = 1;
  });
  
  gtx.strokeStyle = 'rgba(255,0,170,0.3)'; gtx.lineWidth = 2; gtx.setLineDash([10,5]);
  gtx.beginPath(); gtx.moveTo(0,0); gtx.lineTo(0,ARENA_H); gtx.moveTo(ARENA_W,0); gtx.lineTo(ARENA_W,ARENA_H); gtx.moveTo(0,ARENA_H); gtx.lineTo(ARENA_W,ARENA_H); gtx.stroke();
  gtx.setLineDash([]);
  gtx.restore();
}
